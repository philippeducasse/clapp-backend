Production Database Migration Instructions
Prerequisites

Docker containers running (django + db)
SQLite db.sqlite3 file with production data
PostgreSQL configured in settings

Step 1: Run Django Migrations
bashdocker compose -f docker/dockercompose.yml exec django python manage.py migrate
Result: Creates all tables in public schema
Step 2: Load SQLite Data into PostgreSQL
bashdocker compose -f docker/dockercompose.yml exec django python manage.py loaddata data_dump.json
Result: Loads all 875 objects into public schema
Step 3: Connect to PostgreSQL
bashdocker compose -f docker/dockercompose.yml exec -it db /bin/bash
psql -U philippe -d clapp
Step 4: Create Schemas
sqlCREATE SCHEMA template;
CREATE SCHEMA user_2;
Step 5: Create Table Structures in Schemas
sql-- Template schema (organizations)
CREATE TABLE template.venues_venue (LIKE public.venues_venue INCLUDING ALL);
CREATE TABLE template.venues_venuecontact (LIKE public.venues_venuecontact INCLUDING ALL);
CREATE TABLE template.festivals_festival (LIKE public.festivals_festival INCLUDING ALL);
CREATE TABLE template.festivals_festivalcontact (LIKE public.festivals_festivalcontact INCLUDING ALL);
CREATE TABLE template.residencies_residency (LIKE public.residencies_residency INCLUDING ALL);
CREATE TABLE template.residencies_residencycontact (LIKE public.residencies_residencycontact INCLUDING ALL);

-- User_2 schema (private data)
CREATE TABLE user_2.applications_application (LIKE public.applications_application INCLUDING ALL);
CREATE TABLE user_2.applications_application_performances (LIKE public.applications_application_performances INCLUDING ALL);
CREATE TABLE user_2.performances_performance (LIKE public.performances_performance INCLUDING ALL);
CREATE TABLE user_2.performances_dossier (LIKE public.performances_dossier INCLUDING ALL);
Step 6: Copy Data into Schemas
sql-- Copy organizations to template
INSERT INTO template.venues_venue SELECT * FROM public.venues_venue;
INSERT INTO template.venues_venuecontact SELECT * FROM public.venues_venuecontact;
INSERT INTO template.festivals_festival SELECT * FROM public.festivals_festival;
INSERT INTO template.festivals_festivalcontact SELECT * FROM public.festivals_festivalcontact;
INSERT INTO template.residencies_residency SELECT * FROM public.residencies_residency;
INSERT INTO template.residencies_residencycontact SELECT * FROM public.residencies_residencycontact;

-- Copy user_2 private data
INSERT INTO user_2.performances_performance 
SELECT * FROM public.performances_performance WHERE profile_id = 2;

INSERT INTO user_2.performances_dossier 
SELECT * FROM public.performances_dossier 
WHERE performance_id IN (SELECT id FROM public.performances_performance WHERE profile_id = 2);

INSERT INTO user_2.applications_application 
SELECT * FROM public.applications_application WHERE profile_id = 2;

INSERT INTO user_2.applications_application_performances 
SELECT * FROM public.applications_application_performances 
WHERE application_id IN (SELECT id FROM public.applications_application WHERE profile_id = 2);
